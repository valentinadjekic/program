Primjeri programa u C jeziku

Osnovni primjeri

Hello World

Najjednostavniji program koji ispisuje "Hello World" na ekran.

#include <stdio.h>

int main() {
    printf("Hello World!\n");
    return 0;
}
Sabiranje dva broja

Program traži od korisnika da unese dva broja i ispisuje njihov zbir.

#include <stdio.h>

int main() {
    int a, b;
    printf("Unesi prvi broj: ");
    scanf("%d", &a);
    printf("Unesi drugi broj: ");
    scanf("%d", &b);
    printf("Zbir je: %d\n", a + b);
    return 0;
}

Srednji nivo

Provjera par/nepar

Program provjerava da li je uneseni broj paran ili neparan.

#include <stdio.h>

int main() {
    int n;
    printf("Unesi broj: ");
    scanf("%d", &n);
    if (n % 2 == 0)
        printf("Broj je paran.\n");
    else
        printf("Broj je neparan.\n");
    return 0;
}
Fibonaccijev niz

Program ispisuje prvih n brojeva Fibonaccijevog niza.

#include <stdio.h>

int main() {
    int n, t1 = 0, t2 = 1, sljedeci;
    printf("Koliko brojeva? ");
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        printf("%d ", t1);
        sljedeci = t1 + t2;
        t1 = t2;
        t2 = sljedeci;
    }
    printf("\n");
    return 0;
}

Napredni primjeri

Pokazivači i stringovi

Program koristi pokazivač za ispis stringa.

#include <stdio.h>

int main() {
    char *poruka = "Hello, Advanced World!";
    printf("%s\n", poruka);
    return 0;
}
Dinamička alokacija memorije

Program alocira niz cijelih brojeva dinamički i ispisuje ih.

#include <stdio.h>
#include <stdlib.h>

int main() {
    int n;
    printf("Koliko brojeva? ");
    scanf("%d", &n);
    int *niz = (int*)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        niz[i] = i * 2;
        printf("%d ", niz[i]);
    }
    printf("\n");
    free(niz);
    return 0;
}
